Default Power-On State

Pull-up / Pull-down States

Default pull resistor state on BCM2711 at power-on. Can be overridden in software at any time.

Pull-UP (HIGH by default)
Pin reads HIGH when floating. Used for I2C (SDA/SCL), active-LOW signals, buttons with external pull-up. BCM2711 pull-up is ~50 kΩ internally. I2C header pins have board-level 1.8 kΩ pull-ups.
Pull-DOWN (LOW by default)
Pin reads LOW when floating. Majority of GPIOs. Suitable for active-HIGH sensors. Override: GPIO.setup(pin, GPIO.IN, pull_up_down=GPIO.PUD_UP)
⚠ 3.3V Logic Only
ALL GPIO pins are 3.3V logic — NOT 5V tolerant. Connecting 5V directly will damage the BCM2711. Use a level shifter for 5V peripherals.

Code Examples

Quick-start snippets using RPi.GPIO, smbus2, spidev, pyserial, and gpiozero

GPIO — RPi.GPIO
import RPi.GPIO as GPIO
import time

GPIO.setmode(GPIO.BCM)
GPIO.setwarnings(False)

# Output — blink LED on GPIO18
GPIO.setup(18, GPIO.OUT)
for _ in range(5):
    GPIO.output(18, GPIO.HIGH)
    time.sleep(0.5)
    GPIO.output(18, GPIO.LOW)
    time.sleep(0.5)

# Input with pull-up
GPIO.setup(17, GPIO.IN, pull_up_down=GPIO.PUD_UP)
val = GPIO.input(17)  # 1=HIGH, 0=LOW

# Edge interrupt callback
def on_edge(channel):
    print(f"Rising edge on GPIO{channel}")

GPIO.add_event_detect(17, GPIO.RISING, callback=on_edge)

GPIO.cleanup()  # always cleanup
PWM — LED + Servo
import RPi.GPIO as GPIO
import time

GPIO.setmode(GPIO.BCM)
GPIO.setup(18, GPIO.OUT)

# Software PWM @ 1kHz (LED brightness)
pwm = GPIO.PWM(18, 1000)
pwm.start(0)
for dc in range(0, 101, 5):
    pwm.ChangeDutyCycle(dc)
    time.sleep(0.05)
pwm.stop()

# Servo @ 50 Hz — 0° → 180°
servo = GPIO.PWM(18, 50)
servo.start(7.5)     # 90°  neutral
time.sleep(1)
servo.ChangeDutyCycle(2.5)  # 0°
time.sleep(1)
servo.ChangeDutyCycle(12.5) # 180°
time.sleep(1)
servo.stop()
GPIO.cleanup()

# Kernel driver reads sensor files

for sensor in sensors:
    with open(f"{sensor}/w1_slave") as f:
        lines = f.readlines()

        raw = lines[1].split('t=')[1].strip()

I2C — smbus2
import smbus2

bus = smbus2.SMBus(1)  # Bus 1 = header pins

# Scan for all devices
found = []
for addr in range(0x03, 0x78):
    try:
        bus.read_byte(addr)
        found.append(f"0x{addr:02X}")
    except OSError:
        pass
print("Devices:", found)

# Read WHO_AM_I from MPU-6050 (addr 0x68)
who = bus.read_byte_data(0x68, 0x75)
print(f"WHO_AM_I = 0x{who:02X}")  # → 0x68

# Write a register
bus.write_byte_data(0x68, 0x6B, 0x00)  # wake up

bus.close()
SPI — spidev
import spidev

spi = spidev.SpiDev()
spi.open(0, 0)           # bus 0, CE0
spi.max_speed_hz = 500000
spi.mode = 0             # CPOL=0 CPHA=0

# Full-duplex transfer — loopback test
# Wire MOSI (Pin 19) → MISO (Pin 21)
tx = [0xDE, 0xAD, 0xBE, 0xEF]
rx = spi.xfer2(tx)
print(tx == rx)  # True if wired

# Read MCP3008 channel 0 (12-bit ADC)
def read_adc(ch):
    r = spi.xfer2([1, (8 + ch) << 4, 0])
    return ((r[1] & 3) << 8) + r[2]

spi.close()
UART — pyserial
import serial
import time

# Disable serial console first via raspi-config!
ser = serial.Serial(
    port='/dev/serial0',  # UART0 (PL011)
    baudrate=9600,
    bytesize=serial.EIGHTBITS,
    parity=serial.PARITY_NONE,
    stopbits=serial.STOPBITS_ONE,
    timeout=2
)

# Send and receive (loopback: TX→RX)
msg = b'Hello RPi!'
ser.write(msg)
time.sleep(0.1)
response = ser.read(len(msg))
print(response)  # b'Hello RPi!'

ser.close()
