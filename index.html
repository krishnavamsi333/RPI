Default Power-On State

Pull-up / Pull-down States

Default pull resistor state on BCM2711 at power-on. Can be overridden in software.

Pull-UP (HIGH by default)
Pin reads HIGH when floating. Used for I2C (SDA/SCL), active-LOW signals, buttons. BCM2711 pull-up ≈ 50 kΩ. I2C header pins have board-level 1.8 kΩ pull-ups (faster edge rates).
Pull-DOWN (LOW by default)
Pin reads LOW when floating. Suitable for active-HIGH sensors.
Override: GPIO.setup(pin, GPIO.IN, pull_up_down=GPIO.PUD_UP)
⚠ 3.3V Logic ONLY — All GPIO pins
Connecting 5V signals directly will permanently damage the BCM2711 SoC. Use a bi-directional level shifter (e.g., TXS0108E for SPI/I2C) or a voltage divider (resistors) for 5V-to-3.3V read-only signals.

GPIO — RPi.GPIO
import RPi.GPIO as GPIO
import time

GPIO.setmode(GPIO.BCM)
GPIO.setwarnings(False)

# Output — blink LED on GPIO18
GPIO.setup(18, GPIO.OUT)
for _ in range(5):
    GPIO.output(18, GPIO.HIGH)
    time.sleep(0.5)
    GPIO.output(18, GPIO.LOW)
    time.sleep(0.5)

# Input with pull-up
GPIO.setup(17, GPIO.IN, pull_up_down=GPIO.PUD_UP)
val = GPIO.input(17)  # 1=HIGH 0=LOW

# Edge interrupt callback
def on_edge(channel):
    print(f"Rising edge on GPIO{channel}")

GPIO.add_event_detect(17, GPIO.RISING,
    callback=on_edge, bouncetime=200)

GPIO.cleanup()  # always cleanup
PWM — LED + Servo
import RPi.GPIO as GPIO
import time

GPIO.setmode(GPIO.BCM)
GPIO.setup(18, GPIO.OUT)

# Software PWM @ 1kHz (LED fade)
pwm = GPIO.PWM(18, 1000)
pwm.start(0)
for dc in range(0, 101, 5):
    pwm.ChangeDutyCycle(dc)
    time.sleep(0.04)
pwm.stop()

# Servo @ 50 Hz (GPIO18, ALT5)
servo = GPIO.PWM(18, 50)
servo.start(7.5)      # 90° neutral
time.sleep(1)
servo.ChangeDutyCycle(2.5)  # 0°
time.sleep(1)
servo.ChangeDutyCycle(12.5) # 180°
time.sleep(1)
servo.stop()
GPIO.cleanup()
1-Wire — DS18B20
import glob, time

# Requires dtoverlay=w1-gpio in config.txt
def read_temp(sensor_path):
    with open(f"{sensor_path}/w1_slave") as f:
        lines = f.readlines()
    if 'YES' in lines[0]:
        raw = lines[1].split('t=')[1].strip()
        return float(raw) / 1000.0
    return None

sensors = glob.glob('/sys/bus/w1/devices/28-*')
for s in sensors:
    t = read_temp(s)
    name = s.split('/')[-1]
    print(f"{name}: {t:.3f}°C / {t*9/5+32:.1f}°F")
gpiozero — Modern API
from gpiozero import LED, Button, PWMLED, Servo

I2C — smbus2
import smbus2

bus = smbus2.SMBus(1)  # Bus 1 = Pin 3+5

# Scan all 7-bit addresses
found = []
for addr in range(0x03, 0x78):
    try:
        bus.read_byte(addr)
        found.append(f"0x{addr:02X}")
    except OSError:
        pass
print("Devices:", found)

# MPU-6050 (0x68) — read WHO_AM_I
who = bus.read_byte_data(0x68, 0x75)
print(f"WHO_AM_I = 0x{who:02X}")  # → 0x68

# Wake up MPU-6050 (power mgmt reg)
bus.write_byte_data(0x68, 0x6B, 0x00)

# Read 6 bytes (accel X,Y,Z)
data = bus.read_i2c_block_data(0x68, 0x3B, 6)
ax = (data[0] << 8) | data[1]

bus.close()
SPI — spidev
import spidev

spi = spidev.SpiDev()
spi.open(0, 0)          # bus=0, CE0
spi.max_speed_hz = 500000
spi.mode = 0            # CPOL=0 CPHA=0

# Full-duplex loopback (wire MOSI→MISO)
tx = [0xDE, 0xAD, 0xBE, 0xEF]
rx = spi.xfer2(tx)
print(tx == rx)  # True if wired

# MCP3008 ADC — read channel 0
def read_adc(ch):
    assert 0 <= ch <= 7
    r = spi.xfer2([1, (8 + ch) << 4, 0])
    return ((r[1] & 3) << 8) + r[2]

val = read_adc(0)   # 0-1023
voltage = val * 3.3 / 1023
print(f"ADC: {val}  →  {voltage:.3f}V")
spi.close()
UART — pyserial
import serial, time

# Disable serial console via raspi-config first!
ser = serial.Serial(
    port='/dev/serial0',  # UART0 (PL011)
    baudrate=9600,
    bytesize=serial.EIGHTBITS,
    parity=serial.PARITY_NONE,
    stopbits=serial.STOPBITS_ONE,
    timeout=2
)

# Loopback test (bridge TX→RX)
msg = b'Hello RPi!'
ser.write(msg)
time.sleep(0.05)
rx = ser.read(len(msg))
print(rx)  # b'Hello RPi!'

# Non-blocking read loop
ser.timeout = 0
while True:
    if ser.in_waiting:
        line = ser.readline().decode('utf-8', errors='ignore')
        print(line.strip())
    time.sleep(0.01)

ser.close()
Hardware PWM — pigpio
import pigpio, time

# pigpio gives true hardware PWM
# sudo pigpiod  (start daemon first)
pi = pigpio.pi()

# Hardware PWM on GPIO18 (Pin 12)
# freq=50Hz, dutycycle=7.5% (neutral servo)
pi.hardware_PWM(18, 50, 75000)  # 7.5% of 1M
time.sleep(1)

# 0° position (5% = 50000/1000000)
pi.hardware_PWM(18, 50, 50000)
time.sleep(1)

# 180° position (10% = 100000/1000000)
pi.hardware_PWM(18, 50, 100000)
time.sleep(1)

pi.hardware_PWM(18, 0, 0)  # off
pi.stop()
